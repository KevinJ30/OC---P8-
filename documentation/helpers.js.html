/**
 * Get element by css selector
 * 
 * @param {string} selector CSS selector
 * @param {string} scope
 *  
 * @returns HTMLElement selected element
 **/
window.qs = function (selector, scope) {
    return (scope || document).querySelector(selector);
};

/**
 * Get elements by css selector
 * 
 * @param {*} selector CSS selector
 * @param {*} scope 
 * @returns HTMLElement
 **/
window.qsa = function (selector, scope) {
    return (scope || document).querySelectorAll(selector);
};

/**
 * Add Event listener to the target
 * 
 * @param {string} target element to the attach event
 * @param {string} type Event type
 * @param {function} callback event function
 * @param {boolean} useCapture 
 **/
window.$on = function (target, type, callback, useCapture) {
    target.addEventListener(type, callback, !!useCapture);
};

    var targetElement = event.target;
    var potentialElements = window.qsa(selector, target);

/**
 * Attach handler to event for allelements that match the selector
 * 
 * @param {string} target elements to the attach event
 * @param {string} selector CSS Selector
 * @param {string} type Event type
 * @param {function} handler 
 **/
window.$delegate = function (target, selector, type, handler) {
    function dispatchEvent(event) {
        var targetElement = event.target;
        var potentialElements = window.qsa(selector, target);
        var hasMatch = Array.prototype.indexOf.call(potentialElements, targetElement) >= 0;

        if (hasMatch) {
            handler.call(targetElement, event);
        }
    }

    // https://developer.mozilla.org/en-US/docs/Web/Events/blur
    var useCapture = type === 'blur' || type === 'focus';

    window.$on(target, type, dispatchEvent, useCapture);
};

/**
 * Find the element's parent with the given tag name
 * 
 * @param {string} element
 * @param {string} tagName
 **/
window.$parent = function (element, tagName) {
    if (!element.parentNode) {
        return;
    }
    if (element.parentNode.tagName.toLowerCase() === tagName.toLowerCase()) {
        return element.parentNode;
    }
    return window.$parent(element.parentNode, tagName);
};
